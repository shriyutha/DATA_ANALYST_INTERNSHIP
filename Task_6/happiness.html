# Scatter Plot: GDP per Capita vs Happiness Score:

plt.figure(figsize = (10, 5))

plt.scatter(data['GDP per capita'], data['Score'], color = 'black', marker = '.', label = 'Countries')

x = data['GDP per capita']
y = data['Score']
m, b = np.polyfit(x, y, 1)   # slope and intercept
plt.plot(x, m*x + b, linestyle = '--', label = 'Regression Line', color = 'red')

plt.title('GDP per Capita vs Happiness Score (2019)')
plt.xlabel('GDP per Capita')
plt.ylabel('Happiness Score')
plt.grid(True)
plt.show()
